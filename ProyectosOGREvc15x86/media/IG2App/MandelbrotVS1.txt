#version 330 core

//Atributos de los vertices
in vec4 vertex;
in vec2 uv0;

//matriz de transdormacion
uniform mat4 modelViewProjMat;
uniform float tiempo;
out vec2 vUv0;
out vec2 vUv1;

void main(){
    vUv0 = uv0;
    vUv1.s = vUv0.s - 1.2;
    vUv1.t = vUv0.t - 1.2;
    vUv1.s = vUv0.s * tiempo;
    vUv1.t = vUv0.t * tiempo;
    gl_Position = modelViewProjMat * vertex;
}
