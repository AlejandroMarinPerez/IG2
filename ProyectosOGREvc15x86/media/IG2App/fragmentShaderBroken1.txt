#version 330 core

uniform sampler2D textura1; // tipo sampler2D para texturas 2D
uniform vec4 InColor;
uniform vec4 OutColor;
//uniform sampler2D textura2; 
uniform float FB;  // blending factor
uniform float intLuz;

in vec2 vUv0; // out del vertex shader
out vec4 fFragColor; // out del fragment shader

void main() {

    vec3 bujeros1 = vec3(texture(textura1, vUv0));
    if (bujeros1.r > 0.5)
    {
        discard;
    }
    else
    {
        if(gl_FrontFacing)
        {
            OutColor *= intLuz;
            vec4 color = mix(OutColor, FB);
            fFragColor = color;
        }
        else
        {
            InColor *= intLuz;
            vec4 color = mix(InColor, FB);
            fFragColor = color;
        }
    }
}
